Let's use go-func package (github.com/benashford/go-func) just as a example of
type-agnostic realisation of map and filter 
// map:
func stringLength(s string) int {
	return len(s)
}

func doMap() {
    a := []string{"a", "bee", "sea", "dee", "eff", "gee", "aitch"}
    b := Maps(a, stringLength).([]int)
    fmt.Println(b)
}

// filter:
func isEven(a int) bool {
	return a % 2 == 0
}

func doFilter(){
    a := []int{1, 2, 3, 4, 5}
    b := Filters(a, isEven).([]int)
    fmt.Println(b)
}
